// FIRST VALUE0: THE ELEMENT THE BULLET IS MADE OF
// SECOND VALUE1: THE BULLET SPEED
// THIRD VALUE2: RANDOMSHOOTING? (0 = AIM AT MOUSE; 1 = AIM ALL OVER; 2 = AIM AT SPECIFIC ELEMENT)
// FOURTH VALUE3: SHOOT FROM WHAT? (0 = SHOOT FROM MIDDLE OF SCREEN; 1 = SHOOT FROM STARTER ELEMENT)
// FIFTH VALUE4: EXPLODE? (1 = YES; 0 = NO)
// SIXTH VALUE5: IF SHOOTING BY KEY, HOW MANY FRAMES IN BETWEEN SHOTS? (ANY VALUE CAN BE INPUT)
// SEVENTH VALUE6: ADD A SPIRAL ON IT? (0 = NO; ANYTHING ELSE IS CONSIDER THE SPIN RATE) [1 - 100000] (10000 is the best)
// EIGHTH VALUE7: BULLET SIZE? (0 = OFF; ANYTHING ELSE ADD A CIRCLE AND AUTO RESIZES THE PREFIXSpinner) (25-29 are buggy; small sizes are easiest)


// #####################  I M P O R T A N T   S H O O T I N G   L A W S  #####################
TRIGGER PREFIXShootingLaws {
   EXEC PREFIXSetShootingLaws ELEMENT:PREFIXProjectileFire 2 0 0 1 15 10000 10
}

TRIGGER AlwaysOn {
   SET PREFIXINCR (PREFIXINCR + PREFIXSpinning)
   TIMER 1 FRAMES AlwaysOn
}
EXEC AlwaysOn

TRIGGER PREFIXSpinningStuff {
   IF (PREFIXBulletSize >= 1) {
      DRAW PREFIXPREFIXSpinnerELEMENT REPLACEFILLEDCIRCLE PREFIXCIRCX PREFIXCIRCY (PREFIXBulletSize / 3) 0
   } ELSE {
      DRAW PREFIXPREFIXSpinnerELEMENT POINT PREFIXCIRCX PREFIXCIRCY
   }
   SET PREFIXSpinning $0
}


// ~YOU CAN CHANGE ANY VALUES BUT IF YOU CHANGE ANY NAMES CHANGE THEM ALL!!
TRIGGER PREFIXSetShootingLaws {
   SET PREFIXBulletSize 1
   SET PREFIXSpinning $6
   SET PREFIXDelay 1
   SET PREFIXExplodeOnCollide 0
   SET PREFIXAimStyle 2
   SET PREFIXRandomShooting 2
   SET PREFIXBulletSpeed 15
   SET PREFIXBulletElement $0
}

// ~ALSO HAS TOGGLES AND IMPORTANT THINGS, DONT CHANGE
TRIGGER PREFIXShootPREFIXProjectileFire {
   EXEC PREFIXShootingLaws
   IF (PREFIXDelay == 0) {
      EXEC ShotPREFIXDelay (1 RAND 50)
   } ELSE {
      EXEC ShotPREFIXDelay PREFIXDelay
   }
   IF (PREFIXAimStyle == 0) {
      SET PREFIXStartX (WIDTH / 2)
      SET PREFIXStartY (HEIGHT / 2)
   } ELSE {
      SET PREFIXStartX X2
      SET PREFIXStartY Y2
   }
   EXEC PREFIXGET_ANGLE PREFIXStartX PREFIXStartY
   EXEC PREFIXMovePREFIXProjectileFireProjectile (PREFIXStartX * 50) (PREFIXStartY * 50) ((PREFIXtmpX * (PREFIXBulletSpeed * -50)) / tmp) ((PREFIXtmpY * (PREFIXBulletSpeed * -50)) / tmp) PREFIXBulletElement
}

// ATTAINS MANY TOGGLES AND OTHER IMPORTANT THINGS, TRY NOT TO CHANGE NUMBERS AROUND IN HERE
TRIGGER PREFIXMovePREFIXProjectileFireProjectile {
   SET PREFIXAquiredX (($0 + $2) / 50)
   SET PREFIXAquiredY (($1 + $3) / 50)
   SET PREFIXCIRCX (((SIN PREFIXINCR) / (100 - (PREFIXBulletSize * 4))) + PREFIXAquiredX)
   SET PREFIXCIRCY (((COS PREFIXINCR) / (100 - (PREFIXBulletSize * 4))) + PREFIXAquiredY)
   SET PREFIXOffScreen 0
   SET PREFIXImpactCollision 0
   IF (($0 / 50) > WIDTH) <SET PREFIXOffScreen 1>
   IF (($1 / 50) > HEIGHT) <SET PREFIXOffScreen 1>
   IF (($0 / 50) < 0) <SET PREFIXOffScreen 1>
   IF (($1 / 50) < 0) <SET PREFIXOffScreen 1>
   IF (((PREFIXAquiredX + ($2 / 50)) PIXEL (PREFIXAquiredY + ($3 / 50))) INGROUP Stoppers) {
      SET PREFIXOffScreen 1
         IF (PREFIXExplodeOnCollide == 1) <SET PREFIXImpactCollision 1>
   }
   IF ((PREFIXAquiredX PIXEL PREFIXAquiredY) INGROUP Stoppers) {
      SET PREFIXOffScreen 1
         IF (PREFIXExplodeOnCollide == 1) <SET PREFIXImpactCollision 1>
   }
   DRAW PREFIXBulletElement REPLACELINE PREFIXAquiredX PREFIXAquiredY ($2 / 50) ($3 / 50) 0
   DRAW PREFIXBulletElement REPLACELINE (PREFIXAquiredX + 1) (PREFIXAquiredY) ($2 / 50) ($3 / 50) 0
   DRAW PREFIXBulletElement REPLACELINE (PREFIXAquiredX - 1) (PREFIXAquiredY) ($2 / 50) ($3 / 50) 0
   DRAW PREFIXBulletElement REPLACELINE (PREFIXAquiredX) (PREFIXAquiredY + 1) ($2 / 50) ($3 / 50) 0
   DRAW PREFIXBulletElement REPLACELINE (PREFIXAquiredX) (PREFIXAquiredY - 1) ($2 / 50) ($3 / 50) 0
   IF (PREFIXSpinning >= 1) <EXEC PREFIXSpinningStuff PREFIXSpinning>
   IF (PREFIXBulletSize >= 1) <DRAW PREFIXBulletElement REPLACEFILLEDCIRCLE PREFIXAquiredX PREFIXAquiredY PREFIXBulletSize 0>
   IF (PREFIXImpactCollision == 1) <EXEC PREFIXBOBExplosion PREFIXINTERACTION2>
   IF (PREFIXOffScreen == 0) {
      TIMER 1 FRAMES PREFIXMovePREFIXProjectileFireProjectile ($0 + $2) ($1 + $3) $2 $3 $4
   }
}


// THESE TWO TRIGGERS ENABLE THE "RELOAD" SPEED
TRIGGER ShotPREFIXDelay {
   SET PREFIXDelayShooting 1
   TIMER $0 FRAMES PREFIXResumeShot
}
TRIGGER PREFIXResumeShot {
   SET PREFIXDelayShooting 0
}
TRIGGER PREFIXStopShot {
   SET PREFIXOffScreen 1
}

// ATTAINS ANGLE AND PREFIXRandomShooting TOGGLEABLITY
TRIGGER PREFIXGET_ANGLE {
   IF (PREFIXRandomShooting == 1) {
      SET PREFIXtmpX ($0 - (0 RAND WIDTH))
      SET PREFIXtmpY ($1 - (0 RAND HEIGHT))
   }
   IF (PREFIXRandomShooting == 0) {
      SET PREFIXtmpX ($0 - X)
      SET PREFIXtmpY ($1 - Y)
   }
   IF (PREFIXRandomShooting == 2) {
      SET PREFIXtmpX ($0 - PREFIXSHOOTATPREFIXAIMATELEMENTX)
      SET PREFIXtmpY ($1 - PREFIXSHOOTATPREFIXAIMATELEMENTY)
   }
   SET tmp (SQRT ((PREFIXtmpX * PREFIXtmpX) + (PREFIXtmpY * PREFIXtmpY)))
}

// I DONT THINK YOU HAVE TO CHANGE A THING ABOUT THESE CODES BELOW BUT IF YOU DO, DO NOT ALTER ANY NUMBERS (UNLESS YOU KNOW WHAT YOUR DOING), VERY SENSITIVE
TRIGGER PREFIXImpactExplosion {
   IF ($0 INGROUP GROUP:PREFIXIMPACTMATERIAL) <DRAW ELEMENT:PREFIXEXPLOSIONPREFIXDebris FILLEDCIRCLE X1 Y1 5 0>
   FOR explosion1raycount FROM 1 TO 5 DO {
      SET PREFIXYREQ (Y1 * (-2 RAND 2))
      SET PREFIXXREQ (X1 * (-2 RAND 2))
      EXEC PREFIXPREFIXImpactExplosionRay (X1 * 100) (Y1 * 100) PREFIXYREQ PREFIXXREQ 15
   }
}

TRIGGER PREFIXPREFIXImpactExplosionRay {
   DRAW ELEMENT:Clear FILLEDCIRCLE ($0 / 100) ($1 / 100) 2 0
   DRAW PREFIXEXPLOSIONPREFIXSmoke FILLEDCIRCLE ($0 / 100) ($1 / 100) 2 0
   DRAW PREFIXEXPLOSIONPREFIXProjectileFire FILLEDCIRCLE ($0 / 100) ($1 / 100) 2 0
   IF ($4 > 0) <TIMER 1 FRAMES PREFIXPREFIXImpactExplosionRay ($0 - $3) ($1 - $2) ($2 - 1) $3 ($4 - 1)>
}


TRIGGER PREFIXBOBExplosion {
   IF ($0 INGROUP GROUP:PREFIXPREFIXProjectileFireMATERIAL) <DRAW ELEMENT:PREFIXEXPLOSIONPREFIXDebris FILLEDCIRCLE X1 Y1 5 0>   
   FOR PREFIXProjectileFireworkexplosioncount FROM 1 TO 3 DO {
      SET PREFIXYREQ ((PREFIXAquiredY * -1) + (-100 RAND 100))
      SET PREFIXXREQ ((PREFIXAquiredX * -1) + (-100 RAND 100))
      EXEC PREFIXPREFIXBOBExplosionRay (PREFIXAquiredX * 100) (PREFIXAquiredY * 100) PREFIXYREQ PREFIXXREQ 100 3 1
   }
}

TRIGGER PREFIXPREFIXBOBExplosionRay {
   DRAW ELEMENT:Clear FILLEDCIRCLE ($0 / 100) ($1 / 100) 2 0
   DRAW PREFIXEXPLOSIONPREFIXSmoke FILLEDCIRCLE ($0 / 100) ($1 / 100) 2 0
   DRAW PREFIXEXPLOSIONPREFIXProjectileFire FILLEDCIRCLE ($0 / 100) ($1 / 100) 2 0
   IF ($6 > 0) <TIMER 1 FRAMES PREFIXPREFIXBOBExplosionRay ($0 - $3) ($1 - $2) $2 $3 $4 $5 ($6 - 1)> ELSE <FOR PREFIXProjectileFireworkexplosioncount FROM 1 TO 4 DO PREFIXPREFIXBOBExplosionEndRay $0 $1 ((0 - 300) RAND 300) ((0 - 300) RAND 300) 15>
}

TRIGGER PREFIXPREFIXBOBExplosionEndRay {
   DRAW PREFIXEXPLOSIONPREFIXProjectileFire REPLACELINE ($0 / 100) ($1 / 100) ($3 / 50) ($2 / 50)
   IF ($4 > 0) <TIMER 1 FRAMES PREFIXPREFIXBOBExplosionEndRay ($0 - $3) ($1 - $2) ($2 - 1) $3 ($4 - 1)>
}
